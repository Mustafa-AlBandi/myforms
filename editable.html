Your edits are saved online using Firebase and hosted on GitHub Pages.

// Initialize Firebase firebase.initializeApp(firebaseConfig); const db = firebase.firestore(); const editor = document.getElementById("editor"); const docRef = db.collection("pages").doc("main"); // Load saved content from Firestore docRef.get().then((doc) => { if (doc.exists) { editor.innerHTML = doc.data().content || "Start typing here..."; } else { editor.innerHTML = "Start typing here..."; } }); // Save changes in real-time editor.addEventListener("input", () => { docRef.set({ content: editor.innerHTML }); });
